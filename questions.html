Filtering

A: SELECT * FROM cities WHERE allows_drones = true;

A: SELECT * FROM stores WHERE zagat_rating > 7;

A: SELECT name FROM stores WHERE zagat_rating >= 8;

A: SELECT name FROM tacos WHERE vegetarian=true;

A: SELECT name, zagat_rating FROM stores WHERE zagat_rating > 7 OR serves_beer = false;

A: SELECT name, zagat_rating FROM stores WHERE zagat_rating > 8 OR (zagat_rating > 5 AND serves_beer = true);

A: SELECT name, zagat_rating, serves_beer FROM stores WHERE zagat_rating > 8 OR (zagat_rating > 5 AND serves_beer = true);
